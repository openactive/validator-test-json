#### Example for Rules 1, 2 and 3
Not like this:

```javascript
"location": {
  "type": "Location",
  "name": "Harworth and Bircotes Sports Ground, ",
  "address": {
    "type": "PostalAddress",
    "streetAddress": "Scooby Road",
    "addressLocality": "Bircotes",
    "addressRegion": {
      "id": 18,
      "name": "Nottinghamshire",
      "region_id": 4,
      "created_at": "2015-10-19T12:41:59.255Z",
      "updated_at": "2018-05-28T04:47:15.433Z",
      "gmap_location": null
    },
    "postalCode": "DN11 8JT"
  },
  "geo": {
    "type": "GeoCoordinates",
    "latitude": 53.41663,
    "longitude": -1.053956
  }
}
```


Like this:

```javascript
"location": {
  "type": "Location",
  "name": "Harworth and Bircotes Sports Ground", /* Note no trailing comma */
  "address": {
    "type": "PostalAddress",
    "streetAddress": "Scooby Road",
    "addressLocality": "Bircotes",
    "addressRegion": "Nottinghamshire", /* Note this is a string */
    "postalCode": "DN11 8JT",
    "addressCountry": "GB" /* Note this is included */
  },
  "geo": {
    "type": "GeoCoordinates",
    "latitude": 53.41663,
    "longitude": -1.053956
  }
}
```
